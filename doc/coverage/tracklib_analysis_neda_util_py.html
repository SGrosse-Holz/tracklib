7class Loopingtrace: 

8 """ 

9 Trace of looping states 

10 

11 `Loopingtrace` objects associate an integer state with each valid (i.e. 

12 non-``nan``) time point in a given trajectory. To this end, it keeps a list 

13 of those valid time points alongside the list of states. 

14 

15 In the context of looping inference, the states indicate which out of a 

16 list of models to use for evolution up to the labelled point. For the first 

17 data point this means that the associated model specifies the equilibrium 

18 state we start from. 

19 

20 Parameters 

21 ---------- 

22 traj : tracklib.Trajectory 

23 the `Trajectory` to which the `Loopingtrace` is associated. This 

24 information is needed to assemble the list of valid time points. 

25 nStates : int, optional 

26 the number of possible states 

27 thresholds : sized iterable (e.g. list) of float, optional 

28 can be used to initialize the trajectory with states corresponding to a 

29 set of absolute distance thresholds. The state of each frame will be 

30 the number of given thresholds that are smaller than the absolute 

31 distance, i.e. ``thresholds=[5, 10]`` means ``dist <= 5`` is state 0, 

32 ``5 < dist <= 10`` state 1, ``10 < dist`` state 2. Will overwrite 

33 `!nStates` with ``len(thresholds)+1``. 

34 

35 Attributes 

36 ---------- 

37 T : int 

38 total length of the associated `Trajectory` 

39 n : int 

40 total number of states (relevant e.g. for acceptance probabilities) 

41 t : (N,) np.ndarray, dtype=int 

42 valid frames of the associated `Trajectory` 

43 state : (N,) np.ndarray, dtype=int 

44 state associated with each (valid) frame of the trajectory. By default 

45 initialized with zeroes, unless `!thresholds` specified 

46 

47 Notes 

48 ----- 

49 This class forwards the ``Sequence`` interface of its `state` attribute, 

50 i.e. for a `Loopingtrace` ``lt`` ``len(lt)`` gives the number of valid 

51 frames, ``lt[i]`` gives the state associated with the ``i``th valid frame 

52 and ``lt[i] = 2`` can be used to assign that state. Note that this usage is 

53 preferred over accessing ``lt.state``, since it allows for more internal 

54 control, such as type checks upon assignment. 

55 """ 

56 def __init__(self, traj, nStates=2, thresholds=None): 

57 self.T = len(traj) 

58 self.n = nStates 

59 self.t = np.array([i for i in range(len(traj)) if not np.any(np.isnan(traj[i]))]) 

60 self.state = np.zeros((len(self.t),), dtype=int) 

61 

62 if thresholds is not None: 

63 self.n = len(thresholds)+1 

64 dist_traj = traj.abs()[:][self.t, 0] 

65 self.state = np.sum([ \ 

66 (dist_traj > thres).astype(int) \ 

67 for thres in thresholds \ 

68 ], axis=0) 

69 

70 def copy(self): 

71 """ 

72 Copy myself 

73 

74 Creates a copy of the calling object. Faster than 

75 ``deepcopy(loopingtrace)``, i.e. used for performance. 

76 """ 

77 new = self.__new__(type(self)) # Skip init 

78 new.T = self.T 

79 new.n = self.n 

80 new.t = self.t.copy() 

81 new.state = self.state.copy() 

82 return new 

83 

84 def __len__(self): 

85 return len(self.state) 

86 

87 def __getitem__(self, key): 

88 return self.state[key] 

89 

90 def __setitem__(self, key, val): 

91 assert val < self.n 

92 assert isinstance(val, (int, np.integer)) 

93 self.state[key] = val

95 def plottable(self): 

96 """ 

97 Return t, y for plotting as steps 

98 

99 Returns 

100 ------- 

101 t : np.ndarray 

102 y : np.ndarray 

103 

104 Example 

105 ------- 

106 For a `Loopingtrace` ``lt``: 

107 

108 >>> from matplotlib import pyplot as plt 

109 ... plt.plot(*lt.plottable(), color='r') 

110 ... plt.show() 

111 

112 """ 

113 tplot = np.array([np.insert(self.t[:-1], 0, self.t[0]-1), self.t]).T.flatten() 

114 yplot = np.array([self.state, self.state]).T.flatten() 

115 return tplot, yplot

117 def full_valid(self): 

118 """ 

119 Give a full-length array of states 

120 

121 This returns an array of looping states of length equal to the 

122 associated trajectory. Invalid frames in the trajectory (i.e. those 

123 containing ``nan``) will be labelled with the same state as the next 

124 valid one. This implements the interpretation that the state associated 

125 with a frame specifies the model governing the evolution up to that 

126 frame. 

127 

128 Returns 

129 ------- 

130 states : (self.T,) np.ndarray 

131 a valid states associated with each frame in the trajectory, even 

132 the invalid ones. 

133 """ 

134 full = np.zeros((self.T,), dtype=int) 

135 last_ind = -1 

136 for cur_ind, cur_val in zip(self.t, self.state): 

137 full[(last_ind+1):(cur_ind+1)] = cur_val 

138 last_ind = cur_ind 

139 return full
