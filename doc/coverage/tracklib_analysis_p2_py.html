16def P2traj(traj, TA=True, recalculate=False, 

17 function=None, postproc=None, 

18 writeto=DEFKEY, 

19 ): 

20 """ 

21 Calculate a two-point function for a single trajectory 

22 

23 The results are stored in the ``traj.meta[writeto]``. By default, if this 

24 field exists already, the function aborts. To recalculate and overwrite, 

25 set ``recalculate = True``. 

26 

27 Parameters 

28 ---------- 

29 traj : Trajectory 

30 the trajectory for which to calculate the MSD 

31 TA : bool, optional 

32 whether to time average 

33 

34 Other Parameters 

35 ---------------- 

36 recalculate : bool 

37 set to ``True`` to ensure that the calculation is actually performed 

38 function : {'SD', 'D', 'SP'} or callable 

39 the function to evaluate. Should be ``fun(traj[m], traj[n]) --> 

40 float``, where ``m >= n``, and it should be vectorized (i.e. work on 

41 numpy arrays and return the corresponding arrays). 

42 postproc : callable or None 

43 will be applied to the final result, i.e. should take a (T,) array and 

44 return such. 

45 writeto : hashable or None 

46 where to store the output of the calculation in the ``traj.meta`` dict. 

47 Set to ``None`` to return the output dict instead of storing it. 

48 Defaults to ``'P2'``. 

49 

50 Returns 

51 ------- 

52 dict, optional 

53 A dict with keys ``'data', 'N'`` giving the averaged data and the count 

54 of valid data points for each lag time. This is usually written into 

55 the ``traj.meta`` dict, but returned if ``writeto is None``. 

56 

57 See also 

58 -------- 

59 P2dataset, MSD, ACF 

60 

61 Notes 

62 ----- 

63 Explicitly, the ``recalculate`` parameter is equivalent to 

64 >>> del traj.meta[writeto] 

65 """ 

66 if recalculate: 

67 for key in [writeto]: 

68 try: 

69 del traj.meta[key] 

70 except: 

71 pass 

72 

73 if writeto not in traj.meta.keys(): 

74 

75 if TA: 

76 data = [function(traj[:], traj[:])] 

77 data += [function(traj[i:], traj[:-i]) for i in range(1, len(traj))] 

78 else: 

79 istart = np.min(np.nonzero(~np.any(np.isnan(traj.data), (0, 2)))[0]) 

80 data = [[function(traj[i], traj[istart])] for i in range(istart, len(traj))] 

81 

82 with warnings.catch_warnings(): 

83 warnings.filterwarnings(action='ignore', message='Mean of empty slice') 

84 

85 out = { 

86 'N' : np.array([np.count_nonzero(~np.isnan(dat)) for dat in data]), 

87 'data' : np.array([np.nanmean(dat) for dat in data]), 

88 } 

89 if callable(postproc): 

90 out['data'] = postproc(out['data']) 

91 

92 if writeto is None: 

93 return out 

94 else: 

95 traj.meta[writeto] = out

97def P2dataset(dataset, givevar=False, giveN=False, average_in_logspace=False, **kwargs): 

98 """ 

99 Ensemble average two-point functions 

100 

101 Parameters 

102 ---------- 

103 dataset : TaggedSet 

104 a list of `Trajectory` 

105 

106 Other Parameters 

107 ---------------- 

108 givevar : bool, optional 

109 whether to also return the variance around the mean 

110 giveN : bool, optional 

111 whether to return the sample size for each MSD data point 

112 average_in_logspace : bool, optional 

113 set to ``True`` to replace the arithmetic with a geometric mean. 

114 kwargs : keyword arguments 

115 are all forwarded to forwarded to `P2traj`, see that docstring. 

116 

117 Returns 

118 ------- 

119 eP2 : np.ndarray 

120 the calculated ensemble mean 

121 var : np.ndarray, optional 

122 variance around the mean 

123 N : np.ndarray, optional 

124 number of data points going into each estimate 

125 

126 See also 

127 -------- 

128 P2traj, MSD, ACF 

129 """ 

130 try: 

131 p2key = kwargs['writeto'] 

132 except KeyError: 

133 p2key = DEFKEY 

134 

135 for traj in dataset: 

136 P2traj(traj, **kwargs) 

137 

138 P2s = [traj.meta[p2key]['data'] for traj in dataset] 

139 Ns = [traj.meta[p2key]['N'] for traj in dataset] 

140 

141 maxlen = max(len(P2) for P2 in P2s) 

142 allP2 = np.empty((len(P2s), maxlen), dtype=float) 

143 allP2[:] = np.nan 

144 allN = np.zeros((len(Ns), maxlen), dtype=int) 

145 for i, (P2, N) in enumerate(zip(P2s, Ns)): 

146 allP2[i, :len(P2)] = P2 

147 allN[i, :len(N)] = N 

148 allN[np.where(np.isnan(allP2))] = 0 

149 

150 if average_in_logspace: # pragma: no cover 

151 allP2 = np.log(allP2[:, 1:]) 

152 N0 = np.sum(allN[:, 0]) 

153 allN = allN[:, 1:] 

154 

155 N = np.sum(allN, axis=0) 

156 meanN = N / np.sum(allN != 0, axis=0) 

157 with warnings.catch_warnings(): 

158 warnings.filterwarnings(action='ignore', message=r'(invalid value|divide by zero) encountered in true_divide') 

159 eP2 = np.nansum(allP2*allN, axis=0) / N 

160 if givevar: 

161 var = np.nansum((allP2-eP2)**2 * allN, axis=0) / (N-meanN) 

162 

163 if average_in_logspace: # pragma: no cover 

164 eP2 = np.insert(np.exp(eP2), 0, 0) 

165 if givevar: 

166 var = np.insert(np.exp(var), 0, 0) 

167 N = np.insert(N, 0, N0) 

168 

169 if givevar and giveN: # pragma: no cover 

170 return eP2, var, N 

171 elif givevar: 

172 return eP2, var 

173 elif giveN: 

174 return eP2, N 

175 else: 

176 return eP2

178def P2(*args, **kwargs): 

179 """ 

180 Shortcut function to calculate P2s. 

181 

182 Will select either `P2traj` or `P2dataset`, depending on the type of the 

183 first argument. Everything is then forwarded to that function. 

184 

185 This is mostly just a template for the following functions MSD, ACF, ... 

186 

187 See also 

188 -------- 

189 P2traj, P2dataset 

190 """ 

191 if issubclass(type(args[0]), Trajectory): 

192 P2traj(*args, **kwargs) 

193 if 'writeto' in kwargs: 

194 writeto = kwargs['writeto'] 

195 else: 

196 writeto = DEFKEY 

197 return args[0].meta[writeto]['data'] 

198 elif issubclass(type(args[0]), TaggedSet): 

199 return P2dataset(*args, **kwargs) 

200 else: # pragma: no cover 

201 raise ValueError("Did not understand first argument, with type {}".format(type(args[0])))

203def MSD(*args, **kwargs): 

204 def SD(xm, xn): 

205 return np.sum((xm-xn)**2, axis=-1) 

206 

207 return P2(*args, **kwargs, function=SD, writeto='MSD')

209def ACF(*args, **kwargs): 

210 def SP(xm, xn): 

211 return np.sum(xm*xn, axis=-1) 

212 def normalize(data): 

213 return data / data[0] 

214 

215 return P2(*args, **kwargs, function=SP, postproc=normalize, writeto='ACF')
