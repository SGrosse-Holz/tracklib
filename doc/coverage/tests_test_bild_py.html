240 def test_lt_stepping_probability_and_proposal_sample(self): 

241 def num2binlist(num, nbits): 

242 ls = list(bin(num + (1<<nbits)))[3:] # bin() gives '0b1...' 

243 return [int(oi) for oi in ls] 

244 

245 nbit = 6 

246 lts = [bild.Loopingtrace.fromStates(num2binlist(num, nbit), nStates=2) for num in range(1<<nbit)] 

247 

248 for lt_from_ind in np.random.choice(len(lts), 5, replace=False): 

249 self.assertAlmostEqual(np.sum([self.sampler.lt_stepping_probability(lts[lt_from_ind], lt) for lt in lts]), 1, delta=1e-10) 

250 

251 lt_prop = lts[np.random.choice(len(lts))] 

252 p_next = np.array([self.sampler.lt_stepping_probability(lt_prop, lt) for lt in lts]) 

253 ind_best_next = np.argmax(p_next) 

254 lt_best_next = lts[ind_best_next] 

255 N = 100/p_next[ind_best_next] 

256 if N > 10000: # pragma: no cover 

257 raise RuntimeError(f"Don't want to sample {N} profiles") 

258 N_best_next = np.count_nonzero([lt == lt_best_next for lt in self.sampler.lt_gen_proposal_sample_from(lt_prop, nSample=int(N))]) 

259 self.assertAlmostEqual(N_best_next, 100, delta=30) # 3 sigma
