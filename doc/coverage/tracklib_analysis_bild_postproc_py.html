14def logLR_clusterflip(lt, traj, model, prior): 

15 """ 

16 Likelihood ratio for clusterflips 

17 

18 Calculate the log likelihood ratio for clusterflips, i.e. replacing whole 

19 intervals with a different state 

20 

21 Parameters 

22 ---------- 

23 lt : Loopingtrace 

24 the profile to start from 

25 traj, model, prior: Trajectory, Model, Prior 

26 the context for likelihood calculation 

27 

28 Returns 

29 ------- 

30 (N, n) np.ndarray 

31 the likelihood ratios. ``N`` is the number of contiguous intervals in 

32 the `Loopingtrace`, ``n = lt.n - 1`` is the number of alternative 

33 states each interval could have. 

34 """ 

35 def likelihood(lt): 

36 return model.logL(lt, traj) + prior.logpi(lt) 

37 

38 loops = lt.loops(return_='index') 

39 

40 lt_new = lt.copy() 

41 Ls = np.zeros((len(loops), lt.n-1)) 

42 for i, loop in enumerate(loops): 

43 for j, new_state in enumerate(list(range(loop[2])) + list(range(loop[2]+1, lt.n))): 

44 lt_new.state[loop[0]:loop[1]] = new_state 

45 Ls[i, j] = likelihood(lt_new) 

46 

47 lt_new.state[loop[0]:loop[1]] = loop[2] 

48 

49 return Ls - likelihood(lt)

51def optimize_clusterflip(lt, traj, model, prior): 

52 """ 

53 Deterministic optimization using (only) clusterflips 

54 

55 We evaluate `logLR_clusterflip` and perform the (one) best flip it found 

56 (if ``log(LR) > 0``). Then repeat until no improvement possible. 

57 

58 Parameters 

59 ---------- 

60 lt : Loopingtrace 

61 the profile to be optimized 

62 traj, model, prior: Trajectory, Model, Prior 

63 the context for likelihood calculation 

64 

65 Returns 

66 ------- 

67 Loopingtrace 

68 a new, optimized `Loopingtrace` 

69 """ 

70 lt_new = lt.copy() 

71 for _ in range(MAX_ITERATION): 

72 logLR = logLR_clusterflip(lt_new, traj, model, prior) 

73 i, j = np.unravel_index(np.argmax(logLR), logLR.shape) 

74 if logLR[i, j] > 0: 

75 loop = lt_new.loops(return_='index')[i] 

76 lt_new.state[loop[0]:loop[1]] = j if j < loop[2] else j+1 

77 else: 

78 break 

79 else: # pragma: no cover 

80 raise RuntimeError("Exceeded MAX_ITERATION ({})".format(MAX_ITERATION)) 

81 

82 return lt_new

84def logLR_nbit(lt, traj, model, prior, n): 

85 """ 

86 Give likelihood ratios for exhaustive resampling of n-bit moving window 

87 

88 Parameters 

89 ---------- 

90 lt : Loopingtrace 

91 the profile to start from 

92 traj, model, prior: Trajectory, Model, Prior 

93 the context for likelihood calculation 

94 n : int 

95 the window size, i.e. number of contiguous bits to sample exhaustively 

96 

97 Returns 

98 ------- 

99 (N, n) np.ndarray 

100 the likelihood ratios, where ``N = len(lt)-n+1`` and ``n = (lt.n)^n`` (we 

101 insert the value of the unchanged window for completeness). 

102 

103 Note 

104 ---- 

105 This scales as ``O(len(lt) * lt.n ^ n)``, i.e. becomes very expensive very 

106 fast. 

107 """ 

108 def likelihood(lt): 

109 return model.logL(lt, traj) + prior.logpi(lt) 

110 

111 L_ref = likelihood(lt) 

112 

113 lt_new = lt.copy() 

114 Ls = np.zeros((len(lt)-n+1, lt.n**n)) 

115 window_vals = list(itertools.product(range(lt.n), repeat=n)) 

116 

117 for i in range(len(lt)-n+1): 

118 cur_val = deepcopy(lt_new[i:(i+n)]) 

119 cur_j = np.sum(cur_val * lt.n**(np.flip(np.arange(n)))) 

120 

121 for j, window_val in enumerate(window_vals): 

122 if j == cur_j: 

123 Ls[i, j] = L_ref 

124 else: 

125 lt_new.state[i:(i+n)] = window_val 

126 Ls[i, j] = likelihood(lt_new) 

127 

128 lt_new.state[i:(i+n)] = cur_val 

129 

130 return Ls - L_ref

132def optimize_nbit(lt, traj, model, prior, n): 

133 """ 

134 Deterministic optimization using n-bit flips 

135 

136 We evaluate `logLR_nbit` and perform the (one) best flip it found (if 

137 ``log(LR) > 0``). Then repeat until no improvement possible. 

138 

139 Parameters 

140 ---------- 

141 lt : Loopingtrace 

142 the profile to be optimized 

143 traj, model, prior: Trajectory, Model, Prior 

144 the context for likelihood calculation 

145 n : int 

146 the window size, i.e. number of contiguous bits to sample exhaustively 

147 

148 Returns 

149 ------- 

150 Loopingtrace 

151 a new, optimized `Loopingtrace` 

152 """ 

153 lt_new = lt.copy() 

154 for _ in range(MAX_ITERATION): 

155 logLR = logLR_nbit(lt_new, traj, model, prior, n) 

156 i, j = np.unravel_index(np.argmax(logLR), logLR.shape) 

157 

158 if logLR[i, j] > 0: 

159 window_val = np.zeros(n) 

160 for k in range(n): 

161 j, it = divmod(j, lt_new.n) 

162 window_val[n-k-1] = it 

163 

164 lt_new.state[i:(i+n)] = window_val 

165 else: 

166 break 

167 else: # pragma: no cover 

168 raise RuntimeError("Exceeded MAX_ITERATION ({})".format(MAX_ITERATION)) 

169 

170 return lt_new

172def logLR_boundary(lt, traj, model, prior): 

173 """ 

174 Give likelihood ratios for boundary moves 

175 

176 Parameters 

177 ---------- 

178 lt : Loopingtrace 

179 the profile to start from 

180 traj, model, prior: Trajectory, Model, Prior 

181 the context for likelihood calculation 

182 

183 Returns 

184 ------- 

185 (N, 2) np.ndarray 

186 the likelihood ratios, where ``N`` is the number of boundaries within 

187 the profile and the two entries are the likelihood ratios for moving 

188 left and right respectively. 

189 """ 

190 def likelihood(lt): 

191 return model.logL(lt, traj) + prior.logpi(lt) 

192 

193 boundaries = np.nonzero(np.diff(lt.state))[0] # boundary is between b and b+1 

194 if len(boundaries) == 0: 

195 return np.array([]) 

196 

197 lt_new = lt.copy() 

198 Ls = np.zeros((len(boundaries), 2)) 

199 for i, b in enumerate(boundaries): 

200 old_states = deepcopy(lt_new[b:(b+2)]) 

201 

202 # try to move left 

203 lt_new[b] = lt_new[b+1] 

204 Ls[i, 0] = likelihood(lt_new) 

205 lt_new[b] = old_states[0] 

206 

207 # try to move right 

208 lt_new[b+1] = lt_new[b] 

209 Ls[i, 1] = likelihood(lt_new) 

210 lt_new[b+1] = old_states[1] 

211 

212 return Ls - likelihood(lt)

216 Deterministic optimization using boundary moves
