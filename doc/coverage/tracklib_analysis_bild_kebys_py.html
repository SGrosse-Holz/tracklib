22def st2profile(s, theta, traj): 

23 """ s are interval lengths (as fraction of total trajectory length) """ 

24 states = theta*np.ones(len(traj)) 

25 if len(s) > 1: 

26 switchpos = np.cumsum(s)[:-1] 

27 

28 switches = np.floor(switchpos*(len(traj)-1)).astype(int) + 1 # floor(0.0) + 1 = 1 != ceil(0.0) 

29 for i in range(1, len(switches)): 

30 states[switches[i-1]:switches[i]] = theta if i % 2 == 0 else 1-theta 

31 

32 states[switches[-1]:] = theta if len(switches) % 2 == 0 else 1-theta 

33 

34 return bild.Loopingprofile(states)

36def dirichlet_methodofmoments(ss, normalized_weights): 

37 m = normalized_weights @ ss 

38 v = normalized_weights @ (ss - m[np.newaxis, :])**2 

39 s = np.mean(m*(1-m)/v) - 1 

40 return s*m

44def logL(params): 

45 ((s, theta), (traj, model)) = params 

46 profile = st2profile(s, theta, traj) 

47 return model.logL(profile, traj)

49def calculate_logLs(ss, thetas, traj, model): 

50 """ parallel-aware (ordered) """ 

51 todo = zip(ss, thetas) 

52 todo = zip(todo, len(ss)*[(traj, model)]) 

53 imap = parallel._map(logL, todo) 

54 return np.array(list(imap))

56### Proposal distribution ###

58def proposal(a, m, s, theta0): 

59 try: 

60 return ( 

61 stats.dirichlet(a).pdf(s.T) 

62 * ( m*theta0 + (1-m)*(1-theta0) ) 

63 ) 

64 except ValueError: 

65 # dirichlet.pdf got an argument that has a zero somewhere, but its alpha is < 1. 

66 if len(s.shape) == 1: 

67 s = s[None, :] 

68 theta0 = np.asarray(theta0) 

69 

70 ind = np.any(s[:, a < 1] == 0, axis=1) 

71 if np.sum(ind) == 0: 

72 raise RuntimeError("Could not identify 0s in sample") 

73 

74 out = np.empty(len(s), dtype=float) 

75 if np.sum(~ind) > 0: 

76 out[~ind] = proposal(a, m, s[~ind], theta0[~ind]) 

77 out[ind] = np.inf 

78 

79 return out

87def sample_proposal(a, m, N): 

88 ss = stats.dirichlet(a).rvs(N) 

89 thetas = (np.random.rand(N) < m).astype(int) 

90 return ss, thetas

92def fit_proposal(ss, thetas, weights): 

93 weights = weights / np.sum(weights) 

94 a = dirichlet_methodofmoments(ss, weights) 

95 m = thetas @ weights 

96 return a, m

100class FixedkSampler: 

101 # Potential further improvements: 

102 # + make each proposal a mixture of Dirichlet's to catch multimodal behavior 

103 # + improve proposal fitting / braking (better than MOM, maybe more something gradient like?) 

104 def __init__(self, traj, model, 

105 k, N=100, 

106 concentration_brake=1e-2, 

107 polarization_brake=1e-3, 

108 max_fev = 20000, 

109 ): 

110 self.k = k 

111 self.N = N 

112 self.brakes = (concentration_brake, polarization_brake) 

113 

114 self.max_fev = max_fev 

115 self.exhausted = False 

116 

117 self.traj = traj 

118 self.model = model 

119 

120 self.parameters = [(np.ones(k+1), 0.5)] 

121 self.samples = [] # each sample is a list: [s, theta, logL, δ, w], where each entry is an (N, ...) array 

122 self.evidences = [] # each entry: (logev, dlogev, KL) 

123 self.max_logL = -np.inf 

124 

125 # Just sample exhaustively for k = 0, 1 

126 # Note that we assume the parameter space volume n*(n-1)^k * 1 = 2 for n = 2. 

127 # The Dirichlet distribution, being defined on the simplex, assumes a parameter space 

128 # volume of 1/k!. This gives a correction term on the evidence! 

129 # The deltas below should be 1/(Dirichlet parameter space volume) = k!/2. 

130 if self.k == 0: 

131 self._fix_exhaustive([np.ones((2, 1)), np.array([0, 1]), 

132 None, 1/2*np.ones(2), None, 

133 ]) 

134 elif self.k == 1: 

135 switches = np.arange(len(self.traj)-1)/(len(self.traj)-1) 

136 ss = np.array([switches, 1-switches]).T 

137 self._fix_exhaustive([np.concatenate(2*[ss], axis=0), 

138 np.concatenate([np.zeros(len(ss)), np.ones(len(ss))], axis=0), 

139 None, 1/2*np.ones(2*len(ss)), None, 

140 ]) 

141 

142 def _fix_exhaustive(self, sample): 

143 sample[2] = calculate_logLs(sample[0], sample[1], self.traj, self.model) 

144 self.max_logL = np.max(sample[2]) 

145 with np.errstate(under='ignore'): 

146 Ls = np.exp(sample[2] - self.max_logL) 

147 sample[4] = Ls / sample[3] 

148 

149 self.samples.append(sample) 

150 

151 ev_offac = np.mean(sample[4]) # offset by a factor exp(max_logL) 

152 dlogev = 1e-10 

153 logev = np.log(ev_offac) + self.max_logL + np.sum(np.log(np.arange(self.k)+1)) 

154 with np.errstate(divide='ignore'): # we might get log(0), but in that case KL is just +inf, that's fine 

155 KL = logev - np.mean(np.log(sample[4]) + self.max_logL) 

156 self.evidences.append((logev, dlogev, KL)) 

157 

158 # Prevent the actual sampling from running 

159 self.exhausted = True 

160 

161 def step(self): 

162 """ returns False if Sampler is exhausted, True otherwise """ 

163 if self.exhausted: 

164 return False 

165 

166 # Update δ's on old samples 

167 for sample in self.samples: 

168 sample[3] += proposal(*self.parameters[-1], sample[0], sample[1]) 

169 

170 # Put together new sample 

171 sample = 5*[None] 

172 sample[0], sample[1] = sample_proposal(*self.parameters[-1], self.N) 

173 sample[2] = calculate_logLs(sample[0], sample[1], self.traj, self.model) 

174 

175 self.max_logL = max(self.max_logL, np.max(sample[2])) # Keep track of max(logL) such that we can properly convert to weights later 

176 

177 sample[3] = np.zeros(self.N) 

178 for a, m in self.parameters: 

179 sample[3] += proposal(a, m, sample[0], sample[1]) 

180 

181 self.samples.append(sample) 

182 

183 # Calculate weights for all samples 

184 for sample in self.samples: 

185 with np.errstate(under='ignore'): 

186 Ls = np.exp(sample[2] - self.max_logL) 

187 sample[4] = Ls / (sample[3] / len(self.parameters)) 

188 

189 # Update proposal 

190 # full_ensemble is [s, theta, w_offac] 

191 full_ensemble = [np.concatenate([sample[i] for sample in self.samples], axis=0) for i in [0, 1, 4]] 

192 

193 old_a, old_m = self.parameters[-1] 

194 new_a, new_m = fit_proposal(*full_ensemble) 

195 

196 # Keep concentration from exploding 

197 concentration_ratio = np.sum(new_a) / np.sum(old_a) 

198 if np.abs(np.log(concentration_ratio))/self.N > self.brakes[0]: 

199 logfac = self.N*self.brakes[0] 

200 if concentration_ratio < 1: 

201 logfac *= -1 

202 

203 new_a = old_a * np.exp(logfac) 

204 

205 # Keep polarization from exploding 

206 if np.abs(new_m - old_m)/self.N > self.brakes[1]: 

207 if new_m > old_m: 

208 new_m = old_m + self.N*self.brakes[1] 

209 else: 

210 new_m = old_m - self.N*self.brakes[1] 

211 

212 self.parameters.append((new_a, new_m)) 

213 

214 # Evidence & KL 

215 ev_offac = np.mean(full_ensemble[2]) # offset by a factor exp(max_logL) 

216 dlogev = stats.sem(full_ensemble[2]) / ev_offac 

217 logev = np.log(ev_offac) + self.max_logL + np.sum(np.log(np.arange(self.k)+1)) # Note k! correction 

218 

219 with np.errstate(divide='ignore'): # we might get log(0), but in that case KL is just +inf, that's fine 

220 KL = np.log(np.mean(full_ensemble[2])) - np.mean(np.log(full_ensemble[2])) 

221 

222 self.evidences.append((logev, dlogev, KL)) 

223 

224 # Check whether we can still sample more in the future 

225 if len(self.samples)*self.N > self.max_fev: 

226 self.exhausted = True 

227 return True 

228 

229 def t_stat(self, other): 

230 logev0, dlogev0 = self.evidences[-1][:2] 

231 logev1, dlogev1 = other.evidences[-1][:2] 

232 

233 effect = logev0 - logev1 

234 return effect / np.sqrt( dlogev0**2 + dlogev1**2 )

236 def MLE_profile(self): 

237 best_logL = -np.inf 

238 for sample in self.samples: 

239 i = np.argmax(sample[2]) 

240 if sample[2][i] > best_logL: 

241 best_logL = sample[2][i]

249def sample(traj, model, 

250 samples_per_step = 100, 

251 init_runs = 20, 

252 significant_separation_sem_fold = 5, 

253 k_lookahead = 2, 

254 k_max = 20, 

255 sampler_kw = {}, 

256 show_progress=False, 

257 ): 

258 """ uses calculate_logLs, which is parallel-aware (ordered) """ 

259 bar = tqdm(disable = not show_progress) 

260 

261 # Some conditions that come in handy 

262 def all_exhausted(xs): 

263 return all(x.exhausted for x in xs) 

264 def within_lookahead(k, candidates): 

265 return k - np.max([sampler.k for sampler in candidates]) <= k_lookahead 

266 def is_significant(sampler, against, t_thres=significant_separation_sem_fold): 

267 return np.abs(sampler.t_stat(against)) > t_thres 

268 

269 # Initialize 

270 samplers = [] 

271 insignificants = [] 

272 k_updates = [] 

273 

274 # Body of the insignificance resolution 

275 def step_highest_sem(samplers, k_updates=k_updates): 

276 candidates = [sampler for sampler in samplers if not sampler.exhausted] 

277 i_worst = np.argmax([candidate.evidences[-1][1] for candidate in candidates]) 

278 k_updates.append(candidates[i_worst].k) 

279 candidates[i_worst].step() # no need to pay attention to return values, this is done by the loop condition 

280 def get_insignificants(samplers): 

281 # Note on the selection of insignificants: 

282 # we pick them such that the whole set is mutually insignificant, as opposed to say picking everything with 

283 # an insignificant comparison against the best sample. The reason is that if for any k we can find a 

284 # significantly better one, we should use that and forget about the worse one. Also, comparing only to the 

285 # best sample is insane when that happens to have large sem and is exhausted already (which is the 

286 # practical situation that alerted me to this issue). 

287 evidences = np.array([sampler.evidences[-1][0] for sampler in samplers]) 

288 ks = np.argsort(evidences)[::-1] 

289 insignificants = [] 

290 for k in ks: # Note that since any([]) == False the first iteration works fine 

291 if not any(is_significant(samplers[k], other) for other in insignificants): 

292 insignificants.append(samplers[k]) 

293 

294 return insignificants 

295 

296 # Run 

297 for k in range(k_max+1): 

298 samplers.append(FixedkSampler(traj, model, 

299 k=k, N=samples_per_step, 

300 **sampler_kw, 

301 )) 

302 assert len(samplers) == k+1 # Paranoia 

303 

304 # Initial sampling 

305 for _ in range(init_runs): 

306 if not samplers[k].step(): 

307 break 

308 k_updates.append(k) 

309 bar.update() 

310 

311 # Update significances 

312 insignificants = get_insignificants(samplers) 

313 

314 # Insignificance resolution 

315 # Note that we only resolve insignificances if it is "important", i.e. if we have 

316 # exhausted the lookahead already. This ensures that we don't get trapped resolving 

317 # irrelevant insignificances, i.e. those away from the optimum 

318 while (not within_lookahead(k+1, insignificants) 

319 and not all_exhausted(insignificants) 

320 and len(insignificants) > 1 

321 ): 

322 step_highest_sem(insignificants) 

323 bar.update() 

324 insignificants = get_insignificants(samplers) 

325 

326 # If the next iteration would exceed the lookahead, stop 

327 if not within_lookahead(k+1, insignificants): 

328 break 

329 

330 # There might be insignificancies left, in which case we will continue sampling until 

331 # a) they are resolved, or 

332 # b) we exhaust all the samplers 

333 while (len(insignificants) > 1 

334 and not all_exhausted(insignificants) 

335 ): 

336 step_highest_sem(insignificants) 

337 bar.update() 

338 insignificants = get_insignificants(samplers) 

339 

340 bar.close() 

341 

342 # Assemble output 

343 out = {} 

344 out['samplers'] = samplers 

345 out['k_updates'] = k_updates 

346 out['ks'] = np.array([sampler.k for sampler in samplers]) 

347 out['evidence'] = np.array([sampler.evidences[-1][0] for sampler in samplers]) 

348 out['evidence_se'] = np.array([sampler.evidences[-1][1] for sampler in samplers]) 

349 out['best_k'] = np.argmax(out['evidence']) 

350 out['profile'] = samplers[out['best_k']].MLE_profile() 

351 return out

355def logLR_boundaries(profile, traj, model): 

356 def likelihood(profile, traj=traj, model=model): 

357 return model.logL(profile, traj) 

358 

359 boundaries = np.nonzero(np.diff(profile.state))[0] # boundary between b and b+1 

360 if len(boundaries) == 0: 

361 return np.array([]) 

362 

363 profile_new = profile.copy() 

364 Ls = np.empty((len(boundaries), 2)) 

365 Ls[:] = np.nan 

366 for i, b in enumerate(boundaries): 

367 old_states = deepcopy(profile_new[b:(b+2)]) 

368 

369 # move left 

370 profile_new[b] = profile_new[b+1] 

371 Ls[i, 0] = likelihood(profile_new) 

372 profile_new[b] = old_states[0] 

373 

374 # move right 

375 profile_new[b+1] = profile_new[b] 

376 Ls[i, 1] = likelihood(profile_new) 

377 profile_new[b+1] = old_states[1] 

378 

379 return Ls - likelihood(profile)

381def optimize_boundary(profile, traj, model, 

382 max_iteration = 10000, 

383 ): 

384 profile_new = profile.copy() 

385 for _ in range(max_iteration): 

386 logLR = logLR_boundaries(profile_new, traj, model) 

387 if len(logLR) == 0: 

388 break 

389 

390 i, j = np.unravel_index(np.argmax(logLR), logLR.shape) 

391 

392 if logLR[i, j] > 0: 

393 boundaries = np.nonzero(np.diff(profile_new.state))[0] 

394 if ( (j == 0 and boundaries[i] == 0) 

395 or (j == 0 and profile_new[boundaries[i]-1] == profile_new[boundaries[i]+1]) 

396 or (j == 1 and boundaries[i] == len(traj)-2) 

397 or (j == 1 and profile_new[boundaries[i]+2] == profile_new[boundaries[i]]) 

398 ): 

399# raise RuntimeError(f"Trying to abolish boundary at {boundaries[i]}") 

400 return profile_new 

401 

402 profile_new[boundaries[i]+j] = profile_new[boundaries[i]+(1-j)] 

403 else: 

404 break 

405 else: 

406 raise RuntimeError(f"Exceeded max_iteration = {max_iteration}") 

407 

408 return profile_new
