77class RouseModel(Model): 

78 """ 

79 Inference with Rouse models 

80 

81 This inference model uses a given number of `rouse.Model` instances to 

82 choose from for each propagation interval. In the default use case this 

83 switches between a looped and unlooped model, but it could be way more 

84 general than that, e.g. incorporating different looped states, loop 

85 positions, numbers of loop, etc. 

86 

87 Parameters 

88 ---------- 

89 N, D, k : float 

90 parameters for the `rouse.Model`. `!N` is the number of monomers, `!D` 

91 the diffusion constant of a free monomer, `k` the backbone strength. 

92 looppositions : list of 2-tuples of int, optional 

93 list of positions of the extra bond. For each entry, a new 

94 `rouse.Model` instance will be set up. Remember to include an unlooped 

95 model (if wanted) by including a position like ``(0, 0)``. 

96 k_extra : float, optional 

97 the strength of the extra bond. By default equal to `k` 

98 measurement : "end2end" or (N,) np.ndarray 

99 which distance to measure. The default setting "end2end" is equivalent 

100 to specifying a vector ``np.array([-1, 0, ..., 0, 1])``, i.e. measuring 

101 the distance from the first to the last monomer. 

102 

103 Attributes 

104 ---------- 

105 models : list of `rouse.Model` 

106 the models used for inference 

107 

108 Notes 

109 ----- 

110 By default, this model assumes that the difference between the models is 

111 the position of the extra bond. It is easy to generalize this, by editing 

112 the `models` attribute after initialization. The only thing to pay 

113 attention to is that each model needs to have a `!measurement` vector. 

114 

115 The `initial_loopingtrace` for this `Model` is the MLE assuming time scale 

116 separation. I.e. we calculate the timepoint-wise MLE using the exact steady 

117 state distributions of each model. 

118 

119 See also 

120 -------- 

121 Model, rouse.Model 

122 """ 

123 def __init__(self, N, D, k, looppositions=[(0, 0), (0, -1)], k_extra=None, measurement="end2end"): 

124 if k_extra is None: 

125 k_extra = k 

126 if measurement == "end2end": 

127 measurement = np.zeros((N,)) 

128 measurement[0] = -1 

129 measurement[-1] = 1 

130 

131 self.models = [] 

132 for loop in looppositions: 

133 mod = rouse.Model(N, D, k, k_extra, extrabond=loop) 

134 mod.measurement = measurement 

135 self.models.append(mod) 

136 

137 def initial_loopingtrace(self, traj): 

138 # We give the MLE assuming time scale separation 

139 # This is exactly the same procedure as for FactorizedModel, where we 

140 # utilize the steady state distributions of the individual Rouse 

141 # models. 

142 loopingtrace = Loopingtrace.forTrajectory(traj, len(self.models)) 

143 distances = traj.abs()[loopingtrace.t][:, 0] 

144 

145 ss_variances = np.array([mod.measurement @ mod.steady_state(True)[1] @ mod.measurement \ 

146 for mod in self.models]) 

147 ss_likelihoods = -0.5*(distances[:, np.newaxis]**2 / ss_variances[np.newaxis, :] \ 

148 + traj.d*np.log(2*np.pi*ss_variances)[np.newaxis, :]) 

149 

150 loopingtrace.state = np.argmax(ss_likelihoods, axis=1) 

151 return loopingtrace 

152 

153 def logL(self, loopingtrace, traj): 

154 if traj.N == 2: # pragma: no cover 

155 traj = traj.relative() 

156 

157 # if not hasattr(loopingtrace, 'individual_logLs'): 

158 looptrace = loopingtrace.full_valid() 

159 logLs = [rouse.multistate_likelihood(traj[:][:, i], 

160 self.models, 

161 looptrace, 

162 traj.meta['localization_error'][i], 

163 return_individual_likelihoods=True, 

164 )[1] \ 

165 for i in range(traj.d)] 

166 loopingtrace.individual_logLs = np.sum(logLs, axis=0) 

167 

168 return np.nansum(loopingtrace.individual_logLs) 

169 

170 

171 def trajectory_from_loopingtrace(self, loopingtrace, localization_error=0.1, d=3): 

172 arr = np.empty((loopingtrace.T, d)) 

173 arr[:] = np.nan 

174 

175 cur_mod = self.models[loopingtrace[0]] 

176 conf = cur_mod.conf_ss(True, d) 

177 arr[loopingtrace.t[0], :] = cur_mod.measurement @ conf 

178 

179 for i in range(1, len(loopingtrace)): 

180 cur_mod = self.models[loopingtrace[i]] 

181 conf = cur_mod.evolve(conf, True, loopingtrace.t[i]-loopingtrace.t[i-1]) 

182 arr[loopingtrace.t[i], :] = cur_mod.measurement @ conf 

183 

184 return Trajectory.fromArray(arr + localization_error*np.random.normal(size=arr.shape), 

185 localization_error=np.array(d*[localization_error]), 

186 loopingtrace=loopingtrace, 

187 )

189class FactorizedModel(Model): 

190 """ 

191 A simplified model, assuming time scale separation 

192 

193 This model assumes that each point is sampled from one of a given list of 

194 distributions, where there is no correlation between the choice of 

195 distribution for each point. It runs significantly faster than the full 

196 `RouseModel`, but is of course inaccurate if the Rouse time is longer or 

197 comparable to the frame rate of the recorded trajectories. 

198 

199 Parameters 

200 ---------- 

201 distributions : list of distribution objects 

202 these will usually be ``scipy.stats.rv_continuous`` objects (e.g. 

203 Maxwell), but can be pretty arbitrary. The only function they have to 

204 provide is ``logpdf()``, which should take a scalar or vector of 

205 distance values and return a corresponding number of outputs. If you 

206 plan on using `trajectory_from_loopingtrace`, the distributions should 

207 also have an ``rvs()`` method for sampling. 

208 

209 Attributes 

210 ---------- 

211 distributions : list of distribution objects 

212 

213 Notes 

214 ----- 

215 This being a heuristical model, we assume that the localization error is 

216 already incorporated in the `!distributions`, as would be the case if they 

217 come from experimental data. Therefore, this class ignores the 

218 ``meta['localization_error']`` field of `Trajectory`. 

219 

220 Instances of this class memoize trajectories they have seen before. To 

221 reset the memoization, you can either reinstantiate or clear the cache 

222 manually: 

223  

224 >>> model = FactorizedModel(model.distributions) 

225 ... model.clear_memo() 

226 

227 If using ``scipy.stats.maxwell``, make sure to use it correctly, i.e. you 

228 have to specify ``scale=...``. Writing ``scipy.stats.maxwell(5)`` instead 

229 of ``scipy.stats.maxwell(scale=5)`` shifts the distribution instead of 

230 scaling it and leads to ``-inf`` values in the likelihood, which then screw 

231 up the MCMC. The classic error to get for this is ``invalid value 

232 encountered in double_scalars``. This is caused by ``new_logL - cur_logL`` 

233 reading ``- inf + inf`` at the first MCMC iteration, if `logL` returns 

234 ``-inf``. 

235 """ 

236 def __init__(self, distributions): 

237 self.distributions = distributions 

238 self._known_trajs = dict() 

239 

240 def _memo(self, traj): 

241 """ 

242 (internal) memoize `traj` 

243 """ 

244 if not traj in self._known_trajs: 

245 with np.errstate(divide='ignore'): # nans in the trajectory raise 'divide by zero in log' 

246 logL_table = np.array([dist.logpdf(traj.abs()[:][:, 0]) 

247 for dist in self.distributions 

248 ]) 

249 self._known_trajs[traj] = {'logL_table' : logL_table} 

250 

251 def clear_memo(self): 

252 """ 

253 Clear the memoization cache 

254 """ 

255 self._known_trajs = dict() 

256 

257 def initial_loopingtrace(self, traj): 

258 self._memo(traj) 

259 loopingtrace = Loopingtrace.forTrajectory(traj, len(self.distributions)) 

260 loopingtrace.state = np.argmax(self._known_trajs[traj]['logL_table'][:, loopingtrace.t], axis=0) 

261 return loopingtrace 

262 

263 def logL(self, loopingtrace, traj): 

264 self._memo(traj) 

265 return np.sum(self._known_trajs[traj]['logL_table'][loopingtrace.state, loopingtrace.t]) 

266 

267 def trajectory_from_loopingtrace(self, loopingtrace, localization_error=0., d=3): 

268 # Note that the distributions in the model give us only the length, not 

269 # the orientation. So we also have to sample unit vectors 

270 # Furthermore, localization_error should not be added, since 

271 # self.distributions already contain it 

272 arr = np.empty((loopingtrace.T, d)) 

273 arr[:] = np.nan 

274 magnitudes = np.array([self.distributions[state].rvs() for state in loopingtrace.state]) 

275 vectors = np.random.normal(size=(len(magnitudes), d)) 

276 vectors *= np.expand_dims(magnitudes / np.linalg.norm(vectors, axis=1), 1) 

277 arr[loopingtrace.t, :] = vectors 

278 

279 return Trajectory.fromArray(arr, 

280 localization_error=np.array(d*[localization_error]), 

281 loopingtrace=loopingtrace, 

282 )

285def fit(data, modelfamily, **kwargs): 

286 """ 

287 Find the best fit model to a calibration dataset 

288 

289 Parameters 

290 ---------- 

291 data : TaggedSet of Trajectory 

292 the calibration data. Each `Trajectory` should have a `meta 

293 <Trajectory.meta>` entry ``'loopingtrace'`` indicating the true/assumed 

294 `Loopingtrace` for this trajectory. 

295 modelfamily : ParametricFamily of Models 

296 the family of models to consider. 

297 kwargs : kwargs 

298 will be forwarded to `!scipy.optimize.minimize`. We use the defaults 

299 ``method='L-BFGS-B'``, ``maxfun=300``, ``ftol=1e-5``. 

300 

301 Returns 

302 ------- 

303 res : fitresult 

304 the structure returned by `!scipy.optimize.minimize`. The best fit 

305 parameters are ``res.x``, while their covariance matrix can be obtained 

306 as ``res.hess_inv.todens()``. 

307 

308 Examples 

309 -------- 

310 A good measure for relative uncertainty of the estimate is given by 

311 ``(√det(Σ) / Π(x))^(1/n)``, i.e. the major axes of the covariance ellipsoid 

312 over the point estimates, normalized by the dimensionality: 

313 

314 >>> res = neda.models.fit(data, modelfam) 

315 ... relative_uncertainty = ( np.sqrt(np.linalg.det(res.hess_inv.todense())) \ 

316 ... / np.prod(res.x) )**(1/modelfam.nParams) 

317 

318 See also 

319 -------- 

320 ParametricFamily 

321 

322 **Troubleshooting** 

323 - make sure that the magnitude of parameter values is around one. The 

324 minimizer (see `!scipy.optimize.minimize`) defaults to a fixed step 

325 gradient descent, which is useless if parameters are orders of magnitude 

326 bigger than 1. You can also try to play with the minimizer's options to 

327 make it use an adaptive step size. 

328 - make sure units match up. A common mistake is to have a unit mismatch 

329 between localization error and trajectories (e.g. one in μm and the 

330 other in nm). If the localization error is too big (here by a factor of 

331 1000), the fit for `!D` will converge to zero (i.e. ``1e-10``). 

332 - the ``'hess_inv'`` field returned with ``method='L-BFGS-B'`` might not 

333 be super reliable, even if the point estimate is pretty good. Check 

334 initial conditions when using this. 

335 """ 

336 def neg_logL_ds(params): 

337 model = modelfamily.get(*params) 

338 def neg_logL_traj(traj): 

339 return -model.logL(traj.meta['loopingtrace'], traj) 

340 

341 # Parallelize? 

342 return np.nansum(list(map(neg_logL_traj, data))) 

343 

344 minimize_kwargs = { 

345 'method' : 'L-BFGS-B', 

346 'bounds' : modelfamily.bounds, 

347 'options' : {'maxfun' : 300, 'ftol' : 1e-5}, 

348 } 

349 # 'People' might try to override the defaults individually 

350 if not 'options' in kwargs: 

351 for key in minimize_kwargs['options']: 

352 if key in kwargs: 

353 minimize_kwargs['options'][key] = kwargs[key] 

354 del kwargs[key] 

355 minimize_kwargs.update(kwargs) 

356 

357 return scipy.optimize.minimize(neg_logL_ds, modelfamily.start_params, **minimize_kwargs)
