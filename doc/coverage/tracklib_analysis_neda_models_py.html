52class RouseModel(Model): 

53 """ 

54 Inference with Rouse models 

55 

56 This inference model uses a given number of `rouse.Model` instances to 

57 choose from for each propagation interval. In the default use case this 

58 switches between a looped and unlooped model, but it could be way more 

59 general than that, e.g. incorporating different looped states, loop 

60 positions, numbers of loop, etc. 

61 

62 Parameters 

63 ---------- 

64 N, D, k : float 

65 parameters for the `rouse.Model`. `!N` is the number of monomers, `!D` 

66 the diffusion constant of a free monomer, `k` the backbone strength. 

67 looppositions : list of 2-tuples of int, optional 

68 list of positions of the extra bond. For each entry, a new 

69 `rouse.Model` instance will be set up. Remember to include an unlooped 

70 model (if wanted) by including a position like ``(0, 0)``. 

71 k_extra : float, optional 

72 the strength of the extra bond. By default equal to `k` 

73 measurement : "end2end" or (N,) np.ndarray 

74 which distance to measure. The default setting "end2end" is equivalent 

75 to specifying a vector ``np.array([-1, 0, ..., 0, 1])``, i.e. measuring 

76 the distance from the first to the last monomer. 

77 

78 Attributes 

79 ---------- 

80 models : list of `rouse.Model` 

81 the models used for inference 

82 

83 Notes 

84 ----- 

85 By default, this model assumes that the difference between the models is 

86 the position of the extra bond. It is easy to generalize this, by editing 

87 the `models` attribute after initialization. The only thing to pay 

88 attention to is that each model needs to have a `!measurement` vector. 

89 

90 See also 

91 -------- 

92 Model, rouse.Model 

93 """ 

94 def __init__(self, N, D, k, looppositions=[(0, 0), (0, -1)], k_extra=None, measurement="end2end"): 

95 if k_extra is None: 

96 k_extra = k 

97 if measurement == "end2end": 

98 measurement = np.zeros((N,)) 

99 measurement[0] = -1 

100 measurement[-1] = 1 

101 

102 self.models = [] 

103 for loop in looppositions: 

104 mod = rouse.Model(N, D, k, k_extra, extrabond=loop) 

105 mod.measurement = measurement 

106 self.models.append(mod) 

107 

108 def initial_loopingtrace(self, traj): 

109 # TODO: come up with a good scheme here 

110 return Loopingtrace(traj, len(self.models)) 

111 

112 def logL(self, loopingtrace, traj): 

113 if traj.N == 2: # pragma: no cover 

114 traj = traj.relative() 

115 

116 looptrace = loopingtrace.full_valid() 

117 return np.sum([ \ 

118 rouse.multistate_likelihood(traj[:][:, i], 

119 self.models, 

120 looptrace, 

121 traj.meta['localization_error'][i], 

122 ) \ 

123 for i in range(traj.d)])

125class FactorizedModel(Model): 

126 """ 

127 A simplified model, assuming time scale separation 

128 

129 This model assumes that each point is sampled from one of a given list of 

130 distributions, where there is no correlation between the choice of 

131 distribution for each point. It runs significantly faster than the full 

132 `RouseModel`, but is of course inaccurate if the Rouse time is longer or 

133 comparable to the frame rate of the recorded trajectories. 

134 

135 Parameters 

136 ---------- 

137 distributions : list of distribution objects 

138 these will usually be ``scipy.stats.rv_continuous`` objects (e.g. 

139 Maxwell), but can be pretty arbitrary. The only function they have to 

140 provide is ``logpdf()``, which should take a scalar or vector of 

141 distance values and return a corresponding number of outputs. 

142 

143 Attributes 

144 ---------- 

145 distributions : list of distribution objects 

146 

147 Notes 

148 ----- 

149 Instances of this class memoize trajectories they have seen before. To 

150 reset the memoization, you can either reinstantiate or clear the cache 

151 manually: 

152  

153 >>> model = FactorizedModel(model.distributions) 

154 ... model.clear_memo() 

155 

156 If using ``scipy.stats.maxwell``, make sure to use it correctly, i.e. you 

157 have to specify ``scale=...``. Writing ``scipy.stats.maxwell(5)`` instead 

158 of ``scipy.stats.maxwell(scale=5)`` shifts the distribution instead of 

159 scaling it and leads to ``-inf`` values in the likelihood, which then screw 

160 up the MCMC. The classic error to get for this is ``invalid value 

161 encountered in double_scalars``. This is caused by ``new_logL - cur_logL`` 

162 reading ``- inf + inf`` at the first MCMC iteration, if `logL` returns 

163 ``-inf``. 

164 """ 

165 def __init__(self, distributions): 

166 self.distributions = distributions 

167 self._known_trajs = dict() 

168 

169 def _memo(self, traj): 

170 """ 

171 (internal) memoize `traj` 

172 """ 

173 if not traj in self._known_trajs: 

174 with np.errstate(divide='ignore'): # nans in the trajectory raise 'divide by zero in log' 

175 logL_table = np.array([dist.logpdf(traj.abs()[:][:, 0]) 

176 for dist in self.distributions 

177 ]) 

178 self._known_trajs[traj] = {'logL_table' : logL_table} 

179 

180 def clear_memo(self): 

181 """ 

182 Clear the memoization cache 

183 """ 

184 self._known_trajs = dict() 

185 

186 def initial_loopingtrace(self, traj): 

187 self._memo(traj) 

188 loopingtrace = Loopingtrace(traj, len(self.distributions)) 

189 loopingtrace.state = np.argmax(self._known_trajs[traj]['logL_table'][:, loopingtrace.t], axis=0) 

190 return loopingtrace 

191 

192 def logL(self, loopingtrace, traj): 

193 self._memo(traj) 

194 return np.sum(self._known_trajs[traj]['logL_table'][loopingtrace.state, loopingtrace.t])
