12class Prior(metaclass=abc.ABCMeta): 

13 """ 

14 Abstract base class for priors 

15 

16 When subclassing, you need to provide the `logpi` method, which should 

17 return the (log) prior probability for an input `Loopingprofile`. Specifying 

18 `logpi_vectorized` is optional, if you can speed up the prior calculation 

19 over an iterable of loopingprofiles. 

20 """ 

21 @abc.abstractmethod 

22 def logpi(self, profile): 

23 """ 

24 log of prior probability 

25 

26 Parameters 

27 ---------- 

28 profile : Loopingprofile 

29 

30 Returns 

31 ------- 

32 float 

33 

34 See also 

35 -------- 

36 logpi_vectorized 

37 """ 

38 raise NotImplementedError # pragma: no cover 

39 

40 def logpi_vectorized(self, profiles): 

41 """ 

42 Evaluate the prior on multiple loopingprofiles 

43 

44 By default just sequentially evaluates `logpi` on the given 

45 loopingprofiles, which of course does not give a speedup. 

46 

47 Parameters 

48 ---------- 

49 profiles : Sequence (e.g. list) of Loopingprofile 

50 

51 Returns 

52 ------- 

53 np.ndarray, dtype=float 

54 

55 See also 

56 -------- 

57 logpi 

58 """ 

59 return np.array([self.logpi(profile) for profile in profiles])

61class UniformPrior(Prior): 

62 r""" 

63 A uniform prior over loopingprofiles 

64 

65 This is simply :math:`-N\log(n)` for each `Loopingprofile`, where :math:`n` 

66 is the number of states and :math:`N` the number of frames. 

67 """ 

68 def __init__(self, nStates=2): 

69 self.logn = np.log(nStates) 

70 

71 def logpi(self, profile): 

72 return -len(profile)*self.logn

74class GeometricPrior(Prior): 

75 r""" 

76 A geometric prior over #switches in the `Loopingprofile` 

77 

78 Writing :math:`\theta` for the `Loopingprofile` and :math:`k(\theta)` for the 

79 number of switches therein, this is given by 

80 

81 .. math:: \pi(\theta) = \frac{1}{n}(1+(n-1)q)^{N-1} q^{k(\theta)}\,, 

82 

83 where :math:`q\in(0, 1]` is the one parameter of this prior, :math:`n` is 

84 the number of possible states and :math:`N` is the number of (valid) frames 

85 in the trajectory. 

86 

87 Parameters 

88 ---------- 

89 logq : float < 0 

90 log of the parameter q 

91 nStates : int, optional 

92 the number of possible states 

93 

94 Attributes 

95 ---------- 

96 logq : float 

97 n : int 

98 """ 

99 def __init__(self, logq=0, nStates=2): 

100 self.logq = logq 

101 self.n = nStates 

102 

103 self._log_n = np.log(self.n) 

104 with np.errstate(under='ignore'): 

105 self._log_norm_per_dof = np.log(1+np.exp(self.logq)*(self.n-1)) 

106 

107 def logpi(self, profile): 

108 # optimized 

109 return profile.count_switches()*self.logq - (len(profile)-1)*self._log_norm_per_dof - self._log_n 

110 

111 def logpi_vectorized(self, profiles): 

112 profiles = np.array([profile[:] for profile in profiles]) 

113 ks = np.count_nonzero(profiles[:, 1:] != profiles[:, :-1], axis=1) 

114 return ks*self.logq - (profiles.shape[1]-1)*self._log_norm_per_dof - self._log_n

163class NumIntPrior(Prior): 

164 """ 

165 Prior designed to reproduce the effects of Chris' numInt scheme 

166 """ 

167 def __init__(self, numInt, logq=0, nStates=2): 

168 self.K = numInt - 1 

169 self.logq = logq 

170 self.n = nStates 

171 

172 self._lognorm = self.K * np.log(1 + np.exp(self.logq)*(self.n - 1)) + np.log(self.n) 

173 

174 def logpi(self, profile): 

175 k = profile.count_switches() 

176 if k <= self.K: 

177 i = np.arange(k) 

178 N = len(profile) 

179 return k * self.logq + np.log(np.prod((self.K-i)/(N-1-i))) - self._lognorm 

180 else:
