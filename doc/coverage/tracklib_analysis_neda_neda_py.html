1""" 

2Main module of the neda inference package 

3 

4This introduces the `Environment` class, which we use to facilitate repeating 

5tasks like running MCMC, calculating or estimating evidence. Finally, the 

6`main` function runs the whole scheme. Note that both of these are imported 

7into the ``tracklib.analysis.neda`` namespace, i.e. can be imported from there

16class Environment: 

17 """ 

18 Environment for inference runs 

19 

20 This is essentially a semi-disguised way of using global variables for the 

21 `Trajectory` we are looking at, the inference `models.Model` we want to use, and 

22 the `MCMCscheme` together with the `MCMCconfig`. We use this approach 

23 mainly for code readability / useability. 

24 

25 Attributes 

26 ---------- 

27 traj : Trajectory 

28 model : models.Model 

29 MCMCconfig : dict 

30 see `tracklib.util.mcmc.Sampler` 

31 MCMCscheme : type, optional 

32 this should implement the `MCMCScheme` interface (i.e. be a subclass of 

33 this abstract base class). 

34 """ 

35 # Here we use ``env`` instead of ``self``. 

36 def __init__(self, traj, model, MCMCconfig, MCMCscheme=mcmc.TPWMCMC): 

37 self.traj = traj 

38 self.model = model 

39 self.MCMCconfig = MCMCconfig 

40 self.MCMCscheme = MCMCscheme

42 def runMCMC(env, prior): 

43 """ 

44 Run the `MCMCscheme` with a given `prior` 

45 

46 Parameters 

47 ---------- 

48 prior : Prior 

49  

50 Returns 

51 ------- 

52 mcmc.MCMCRun 

53 """

57 return mc.run()

59 def posterior_density(env, mcmcrun, prior, trace_eval, 

60 nSample_proposal=float('inf')): 

61 r""" 

62 Estimate the posterior density at a specific point from an `MCMCRun` 

63 

64 Parameters 

65 ---------- 

66 mcmcrun : MCMCRun 

67 prior : Prior 

68 trace_eval : Loopingtrace 

69 the point in parameter space for which to estimate the posterior 

70 density 

71 nSample_proposal : float, optional 

72 how many samples from the proposal distribution to use for 

73 evaluation of the exit rate. Defaults to exhaustive sampling, so 

74 may be reduced if that seems excessive. 

75 

76 Returns 

77 ------- 

78 float 

79 the estimated posterior density for `!trace_eval`. Might be 

80 ``np.inf`` if the MCMC ensemble collapsed to a single sample. 

81 

82 Notes 

83 ----- 

84 This estimation follows eq. (9) of [1]_. Essentially, we estimate the 

85 density at the sample point as the number of times it is entered from 

86 another point in the steady state ensemble, times the average number of 

87 steps it takes to leave this point again, divided by the total sample 

88 number :math:`M`. This yields 

89 

90 .. math:: \hat{p}(\theta^*) = \frac{N_\text{enter}N_\text{stay}}{M} = \frac{P_\text{enter}}{k_\text{exit}} = \frac{M^{-1}\sum_{m=1}^M\, \alpha(\theta^{(m)}, \theta^*) q(\theta^{(m)}, \theta^*)}{J^{-1}\sum_{j=1}^J\, \alpha(\theta^*, \theta^{(j)})}\,, 

91 

92 where the :math:`\theta^{(m)}` in the numerator are the samples from 

93 the given MCMC run (i.e. are assumed to be sampled from the posterior 

94 distribution), while :math:`\theta^{(j)}` in the denominator are 

95 independent samples from the proposal distribution 

96 :math:`q(\theta_\text{from} = \theta^*, \theta_\text{to})` 

97 around the evaluation point :math:`\theta^*`. Finally, 

98 :math:`\alpha(\theta_\text{from}, \theta_\text{to})` is the 

99 acceptance probability for a given step. 

100 

101 References 

102 ---------- 

103 .. [1] Chib, S. & Jeliazkov, I. Marginal Likelihood From the Metropolis-Hastings Output. Journal of the American Statistical Association 96, 270-281 (2001) 

104 """ 

105 # Check whether the MCMC sample collapsed, in which case the posterior 

106 # estimation would diverge 

107 logLs = mcmcrun.logLs_trunc() 

108 if np.all(logLs == logLs[0]): 

109 print('MCMC ensemble collapsed to single sample') 

110 return np.inf 

111 

112 L_eval = env.MCMCscheme.likelihood(env.traj, trace_eval, env.model, prior) 

113 

114 # Chance to enter the target state from steady state 

115 def get_p_step_to_eval(trace): return env.MCMCscheme.stepping_probability(trace, trace_eval) 

116 p_step_to_eval = np.array(mcmcrun.evaluate(get_p_step_to_eval)) 

117 

118 # Note: p_accept = 1 *if* we evaluate at the maximum likelihood trace 

119 p_accept_move = env.MCMCscheme.acceptance_probability(mcmcrun.logLs_trunc(), L_eval) 

120 p_enter = np.mean(p_step_to_eval * p_accept_move) 

121 

122 # Survival in target state 

123 neighbor_logLs = [env.MCMCscheme.likelihood(env.traj, trace, env.model, prior) \ 

124 for trace in env.MCMCscheme.gen_proposal_sample_from(trace_eval, nSample=nSample_proposal)] 

125 k_leave = np.mean(env.MCMCscheme.acceptance_probability(L_eval, neighbor_logLs)) 

126 

127 # Occupancy of the target state = p_enter * survival 

128 return p_enter / k_leave

130 def evidence(env, prior, mcmcrun): 

131 """ 

132 Calculate the (log-)evidence from an `MCMCRun` 

133 

134 Parameters 

135 ---------- 

136 prior : Prior 

137 the prior over `Loopingtraces <Loopingtrace>` used for the MCMC run 

138 mcmcrun : MCMCRun 

139 the MCMC sample 

140 

141 Returns 

142 ------- 

143 float 

144 the estimated log-evidence 

145 

146 Notes 

147 ----- 

148 The log-evidence is given by ``log(likelihood) + log(prior) - 

149 log(posterior)``. 

150 

151 See also 

152 -------- 

153 posterior_density 

154 """ 

155 trace_eval, L_eval = mcmcrun.best_sample_L() 

156 log_post = np.log(env.posterior_density(mcmcrun, prior, trace_eval)) 

157 log_prior = prior.logpi(trace_eval) 

158 

159 if np.isinf(log_post): 

160 return L_eval 

161 else: 

162 return L_eval + log_prior - log_post

164 def evidence_differential(env, prior, ref_prior, ref_mcmcrun): 

165 r""" 

166 Estimate the relative evidence given a reference point 

167 

168 Parameters 

169 ---------- 

170 prior : Prior 

171 the prior whose evidence we want to estimate 

172 ref_prior : Prior 

173 the reference point in prior space 

174 ref_mcmcrun : MCMCRun 

175 an `MCMCRun` using the reference prior 

176 

177 Returns 

178 ------- 

179 float 

180 the estimated log-evidence, relative to the reference 

181 

182 Notes 

183 ----- 

184 The relative evidence is given by the expectation value of the prior 

185 ratio over the MCMC sample: ``E = < prior/ref_prior >`` where ``<.>`` 

186 indicates an average over the MCMC sample. 

187 """ 

188 return np.log(np.mean(np.exp(prior.logpi_vectorized(ref_mcmcrun.samples) - \ 

189 ref_prior.logpi_vectorized(ref_mcmcrun.samples))))

191def main(traj, model, priorfac, 

192 MCMCconfig, MCMCscheme=mcmc.TPWMCMC, 

193 max_iterations=20, min_iterations=5, 

194 return_ = 'nothing', # 'traj', 'dict', or anything else 

195 show_progress=False, assume_notebook_for_progressbar=True, 

196 ): 

197 """ 

198 Run the neda looping inference scheme 

199 

200 The output of the inference run will be assembled into a dict whose fields 

201 are detailed below. Where exactly this dict will end up depends on the 

202 setting of `!return_`. 

203  

204 Parameters 

205 ---------- 

206 traj : Trajectory 

207 the `Trajectory` whose looping profile to infer 

208 model : models.Model 

209 the inference model to use 

210 priorfac : PriorFactory 

211 a family of priors 

212 MCMCconfig : dict 

213 configuration for the MCMC runs. See `tracklib.util.mcmc.Sampler` 

214 MCMCscheme : type, optional 

215 a class implementing the `MCMCScheme` interface (i.e. a subclass of 

216 this abstract base class). Defines the MCMC sampling scheme 

217 max_iterations : int, optional 

218 maximum number of iterations to run 

219 min_iterations : int, optional 

220 run at least this many iterations 

221 return_ : {'nothing', 'None', 'traj', 'dict'} 

222 what the return value of this function should be. Generally, the 

223 results of the inference run will be stored in a dict. If 

224 ``return_='dict'``, that dict is directly returned. Otherwise it is 

225 written to ``traj.meta['neda']``, and if ``return_='traj'`` the 

226 trajectory is returned (useful for parallelization). Otherwise this 

227 function returns nothing, i.e. the inference results can be accessed 

228 simply from ``traj.meta['neda']`` after calling this function. 

229 show_progress : bool, optional 

230 whether to show a progress bar. Note: for ``show_progress=True`` it 

231 might happen that the termination condition is fulfilled during the 

232 last "required" run, in which case the progress bar will stop before 

233 reaching its maximum. 

234 assume_notebook_for_progressbar : bool, optional 

235 set to ``False`` if running outside Jupyter notebook to show 

236 progressbar ASCII style 

237 

238 Returns 

239 ------- 

240 prior_prams : np.array 

241 the prior parameters for each iteration 

242 mcmcrun : list of MCMCRun 

243 the MCMC runs for each iteration 

244 evidence : np.array 

245 the evidence estimated from each `MCMCRun` 

246 evidence_diff : np.array 

247 the estimated evidence differential (relative evidence) at each 

248 iteration 

249 final : dict 

250 the corresponding values for the iteration that should be considered 

251 the final result. Has entries ``'prior_params'``, ``'mcmcrun'``, 

252 ``'evidence'``, and ``'iteration'`` where the last one is the index of 

253 the iteration the other three refer to. 

254 

255 Example 

256 ------- 

257 Assuming we have a `Trajectory` ``traj`` that we want to run the looping 

258 inference on: 

259 

260 >>> # Set up inference scheme 

261 ... looppositions = [(0, 0), (0, -1)] # 2 states: unlooped (=0), fully looped (=1) 

262 ... model = neda.models.RouseModel(N=20, D=1, k=5, k_extra=1, looppositions=looppositions) 

263 ... priorfac = neda.priors.PriorFactory(start_params=(0), bounds=[(None, 0)]) 

264 ... priorfac.get = lambda logq : neda.priors.GeometricPrior(logq, nStates=len(looppositions)) 

265 ... MCMCconfig = { 

266 ... 'iterations' : 1000, 

267 ... 'burn_in' : 100, 

268 ... } 

269 ...  

270 ... # Run the inference 

271 ... neda.main(traj, model, priorfac, MCMCconfig, show_progress=True) 

272 ...  

273 ... # Visualize output 

274 ... from matplotlib import pyplot as plt 

275 ... neda.plot.butterfly(traj) 

276 ... plt.show() 

277 

278 """ 

279 assert min_iterations >= 2 

280 assert return_ in {'nothing', 'None', 'traj', 'dict'} 

281 

282 # Set up environment 

283 env = Environment(traj, model, MCMCconfig, MCMCscheme) 

284 

285 # Set up iterative scheme 

286 iterations = range(max_iterations) 

287 if show_progress: # pragma: no cover 

288 if assume_notebook_for_progressbar: 

289 from tqdm.notebook import tqdm 

290 else: 

291 from tqdm import tqdm 

292 iterations = tqdm(iterations, total=min(min_iterations, max_iterations)) 

293 

294 prior_params = [priorfac.start_params] 

295 mcmcruns = [] 

296 evidences = [] 

297 evidence_diffs = [] 

298 

299 # Run iterations 

300 for it in iterations: 

301 prior = priorfac.get(*prior_params[-1]) 

302 mcmcruns.append(env.runMCMC(prior)) 

303 evidences.append(env.evidence(prior, mcmcruns[-1])) 

304 

305 if it+1 >= min_iterations: 

306 if np.argmax(evidences) < it: 

307 break 

308 

309 # Maximize estimated evidence differential to find new prior parameters 

310 def minimization_target(*params): 

311 return -env.evidence_differential(priorfac.get(*params), prior, mcmcruns[-1]) 

312 minimization_result = scipy.optimize.minimize(minimization_target, 

313 x0=prior_params[-1], 

314 bounds=priorfac.bounds) 

315 

316 if not minimization_result.success: # pragma: no cover 

317 print(minimization_result) 

318 raise RuntimeError('Relative evidence maximization did not converge') 

319 

320 evidence_diffs.append(-minimization_result.fun) 

321 prior_params.append(tuple(minimization_result.x)) 

322 

323 # Output 

324 best_it = np.argmax(evidences) 

325 output = { 

326 'prior_params' : np.array(prior_params), 

327 'mcmcrun' : mcmcruns, 

328 'evidence' : np.array(evidences), 

329 'evidence_diff' : np.array(evidence_diffs), 

330 'final' : { 

331 'prior_params' : prior_params[best_it], 

332 'mcmcrun' : mcmcruns[best_it], 

333 'evidence' : evidences[best_it], 

334 'iteration' : best_it, 

335 }, 

336 } 

337 

338 if return_ == 'dict': 

339 return output 

340 else: 

341 traj.meta['neda'] = output 

342 if return_ == 'traj': 

343 return traj 

344 else: 

345 return None
