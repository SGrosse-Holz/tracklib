15class Prior(metaclass=abc.ABCMeta): 

16 """ 

17 Abstract base class for priors 

18 

19 When subclassing, you need to provide the `logpi` method, which should 

20 return the (log) prior probability for an input `Loopingtrace`. Specifying 

21 `logpi_vectorized` is optional, if you can speed up the prior calculation 

22 over an iterable of loopingtraces. 

23 """ 

24 @abc.abstractmethod 

25 def logpi(self, loopingtrace): 

26 """ 

27 log of prior probability 

28 

29 Parameters 

30 ---------- 

31 loopingtrace : Loopingtrace 

32 

33 Returns 

34 ------- 

35 float 

36 

37 See also 

38 -------- 

39 logpi_vectorized 

40 """ 

41 raise NotImplementedError # pragma: no cover 

42 

43 def logpi_vectorized(self, loopingtraces): 

44 """ 

45 Evaluate the prior on multiple loopingtraces 

46 

47 By default just sequentially evaluates `logpi` on the given 

48 loopingtraces, which of course does not give a speedup. 

49 

50 Parameters 

51 ---------- 

52 loopingtraces : Sequence (e.g. list) of Loopingtrace 

53 

54 Returns 

55 ------- 

56 np.ndarray, dtype=float 

57 

58 See also 

59 -------- 

60 logpi 

61 """ 

62 return np.array([self.logpi(trace) for trace in loopingtraces])

64class PriorFactory: 

65 """ 

66 A factory class for priors 

67 

68 This class encodes usage patterns for priors. To that end, it stores a 

69 reasonable set of default/initial parametrs for the given prior, as well as 

70 the domains for each prior. It then also provides a method to convert a set 

71 of parameters to a `Prior` object. In summary, this class provides 

72 everything we need to run a generic fitting algorithm over priors. 

73 

74 Parameters 

75 ---------- 

76 start_params : tuple of float 

77 the initial/default parameters for a given class of priors 

78 bounds : list of (lower, higher) tuples 

79 domain for each parameter 

80 

81 Attributes 

82 ---------- 

83 start_params : tuple of float 

84 nParams : int 

85 bounds : list of (lower, higher) tuples 

86 

87 Example 

88 ------- 

89 A factory for `GeometricPrior` (with 2 states): 

90 

91 >>> priorfac = PriorFactory((0,), [(None, 0)]) 

92 ... priorfac.get = lambda logq : GeometricPrior(logq) 

93 ... priorfac.get = GeometricPrior # also works in this special case, but suboptimal style 

94  

95 Note that `GeometricPrior` also has a `factory` method providing 

96 essentially this construction. 

97 

98 See also 

99 -------- 

100 Prior 

101 """ 

102 def __init__(self, start_params, bounds): 

103 self.start_params = start_params 

104 self.nParams = len(start_params) 

105 self.bounds = bounds 

106 

107 def get(self, *params): 

108 """ 

109 Generate a prior from the given parameters 

110 

111 This function should be overwritten upon instantiation 

112 """ 

113 raise NotImplementedError # pragma: no cover

115class UniformPrior(Prior): 

116 r""" 

117 A uniform prior over loopingtraces 

118 

119 This is simply :math:`-N\log(n)` for each `Loopingtrace`, where :math:`n` 

120 is the number of states and :math:`N` the number of (valid) frames. 

121 """ 

122 def logpi(self, loopingtrace): 

123 return -len(loopingtrace.t)*np.log(loopingtrace.n)

125class GeometricPrior(Prior): 

126 r""" 

127 A geometric prior over #switches in the `Loopingtrace` 

128 

129 Writing :math:`\theta` for the `Loopingtrace` and :math:`k(\theta)` for the 

130 number of switches therein, this is given by 

131 

132 .. math:: \pi(\theta) = \frac{1}{n}(1+(n-1)q)^{N-1} q^{k(\theta)}\,, 

133 

134 where :math:`q\in(0, 1]` is the one parameter of this prior, :math:`n` is 

135 the number of possible states and :math:`N` is the number of (valid) frames 

136 in the trajectory. 

137 

138 Parameters 

139 ---------- 

140 logq : float < 0 

141 log of the parameter q 

142 nStates : int, optional 

143 the number of possible states 

144 

145 Attributes 

146 ---------- 

147 logq : float 

148 n : int 

149 """ 

150 def __init__(self, logq=0, nStates=2): 

151 self.logq = logq 

152 self.n = nStates 

153 

154 self._log_n = np.log(self.n) 

155 self._log_norm_per_dof = np.log(1+np.exp(self.logq)*(self.n-1)) 

156 

157 def logpi(self, loopingtrace): 

158 # optimized 

159 k = np.count_nonzero(loopingtrace.state[1:] != loopingtrace.state[:-1]) 

160 return k*self.logq - (len(loopingtrace.t)-1)*self._log_norm_per_dof - self._log_n 

161 

162 def logpi_vectorized(self, loopingtraces): 

163 loopingtraces = np.array([trace.state for trace in loopingtraces]) 

164 ks = np.count_nonzero(loopingtraces[:, 1:] != loopingtraces[:, :-1], axis=1) 

165 return ks*self.logq - (loopingtraces.shape[1]-1)*self._log_norm_per_dof - self._log_n

167 @classmethod 

168 def factory(cls, nStates=2): 

169 """ 

170 Give a `PriorFactory` for `GeometricPrior` 

171 

172 Parameters 

173 ---------- 

174 nStates : int, optional 

175 the number of states the priors will assume 

176 

177 Returns 

178 ------- 

179 PriorFactory 

180 

181 See also 

182 -------- 

183 PriorFactory 

184 """ 

185 fac = PriorFactory((0,), [(None, 0)]) 

186 fac.get = lambda logq : cls(logq, nStates) 

187 return fac
