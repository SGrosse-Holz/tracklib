12def evalSPT(filename, tags=set()): 

13 """ 

14 Load a file in the format used by evalSPT. 

15  

16 Parameters 

17 ---------- 

18 filename : string 

19 the file to be read 

20 tags : str, list of str or set of str, optional 

21 the tag(s) to be associated with trajectories from this file 

22 

23 Returns 

24 ------- 

25 TaggedSet 

26 the loaded data set 

27 

28 Notes 

29 ----- 

30 **Format specification:** the file handed to this function should be a text 

31 file, delimited by tabs, and without header line. The columns are 

32 interpreted as ``[x, y, t, id, ...]``, where ``...`` indicates that 

33 additional columns in the data will be ignored. The time is assumed to be 

34 given in frames, i.e. should be an integer, but can start at an arbitrary 

35 number.. The `!id` column should identify trajectories, i.e. all the data 

36 belonging to one trajectory should be consecutive and have the same `!id`. 

37 `!id` do not have to start at ``1``, or be consecutive integers. 

38 """ 

39 def data2gen(data, tags): 

40 for mydata in np.split(data, np.where(np.diff(data[:, 3]) != 0)[0]+1): 

41 t = mydata[:, 2].astype(int) 

42 t -= np.min(t) 

43 traj = np.empty((np.max(t)+1, 2)) 

44 traj[:, :] = np.nan 

45 traj[t, :] = mydata[:, :2] 

46 yield (Trajectory.fromArray(traj), tags) 

47 

48 try: 

49 data = np.genfromtxt(filename, delimiter='\t') 

50 except Exception as err: 

51 print("Error loading file '{}'".format(filename)) 

52 raise err 

53 return TaggedSet(data2gen(data, tags))
