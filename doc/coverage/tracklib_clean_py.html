10def split_trajectory_at_big_steps(traj, threshold): 

11 """ 

12 Removes suspected mislinkages. 

13 

14 Exception: two consecutive big steps such that the overall displacement is 

15 smaller than the threshold are single misconnections in otherwise fine 

16 trajectories. In this case we will simply remove the misconnected point. 

17 

18 Parameters 

19 ---------- 

20 traj : Trajectory 

21 the trajectory to investigate 

22 threshold : float 

23 the maximum allowed frame to frame displacement 

24  

25 Returns 

26 ------- 

27 set of tracklib.Trajectory 

28 

29 See also 

30 -------- 

31 split_dataset_at_big_steps 

32 

33 Notes 

34 ----- 

35 As of now, this only works on trajectories with ``N=1``. 

36 

37 This really just checks for frame-to-frame connections exceeding the 

38 threshold. So if there are missing frames in a trajectory, the step across 

39 those missing data will not be considered. 

40 """ 

41 if traj.N != 1: # pragma: no cover 

42 raise ValueError("Cannot detect mislinkages in trajectories with N > 1") 

43 

44 old_npwarns = np.seterr(invalid='ignore') # yes, we'll do np.nan <= threshold. Gives False. 

45 

46 difftraj = traj.diff().abs() 

47 step_isBig = np.where(difftraj[:][:, 0] <= threshold, 0, 1.) 

48 step_isBig[np.where(np.isnan(difftraj[:][:, 0]))[0]] = np.nan 

49 

50 np.seterr(**old_npwarns) 

51 

52 step_isBig = np.pad(step_isBig, 1, constant_values=np.nan) # Now step_isBig[i] describes traj[i] - traj[i-1] 

53 inds_bigsteps = np.where(step_isBig == 1)[0] 

54 

55 # Check for single misconnections 

56 for ind in inds_bigsteps: 

57 if ((step_isBig[(ind-1):(ind+2)] == 1).tolist() == [False, True, True] 

58 and step_isBig[ind+2] != 1): 

59 traj.data[:, ind, :] = np.nan 

60 step_isBig[ind:(ind+2)] = np.nan # Now the steps don't exist anymore 

61 

62 # If now everything's fine, that's cool 

63 if not np.any(step_isBig == 1): 

64 return {traj} 

65 

66 # Split at remaining big steps 

67 inds_bigsteps = np.where(step_isBig == 1)[0] 

68 new_trajs = set() 

69 old_ind = 0 

70 for ind in inds_bigsteps: 

71 new_trajs.add(Trajectory.fromArray(traj.data[:, old_ind:ind, :])) 

72 old_ind = ind 

73 new_trajs.add(Trajectory.fromArray(traj.data[:, old_ind:, :])) 

74 del traj 

75 

76 return {traj for traj in new_trajs if len(traj) > 1}

78def split_dataset_at_big_steps(data, threshold): 

79 """ 

80 Apply `split_trajectory_at_big_steps` to a whole data set 

81 

82 Parameters 

83 ---------- 

84 data : TaggedSet 

85 the dataset 

86 threshold : float 

87 the maximum allowed step size 

88 

89 Returns 

90 ------- 

91 TaggedSet 

92 a new data set with the split trajectories 

93 

94 See also 

95 -------- 

96 split_trajectory_at_big_steps 

97 """ 

98 def gen(): 

99 for traj, tags in data(giveTags=True): 

100 for part_traj in split_trajectory_at_big_steps(traj, threshold): 

101 yield (deepcopy(part_traj), deepcopy(tags)) 

102 

103 return TaggedSet(gen())
