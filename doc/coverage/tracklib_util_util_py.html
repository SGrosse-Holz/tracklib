13def _twoLociRelativeACF(ts, A=1, B=1, d=1): # pragma: no cover 

14 """ 

15 A = σ^2 / √κ (general prefactor) 

16 B = Δs^2 / 4κ (tether length) 

17 """ 

18 # Scipy's implementation of En can only deal with integer n 

19 def E32(z): 

20 return 2*np.exp(-z) - 2*np.sqrt(np.pi*z)*scipy.special.erfc(np.sqrt(z)) 

21 

22 if not isinstance(ts, collections.abc.Iterable): 

23 ts = [ts] 

24 

25 return np.array([ d*A*( np.sqrt(B) - np.sqrt(t/np.pi)*( 1 - 0.5*E32(B/t) ) ) if t != 0 else d*A*np.sqrt(B) for t in ts])

28 """ 

29 A = σ^2 / √κ (general prefactor)

35def distribute_noiselevel(noise2, pixelsize): 

36 """ 

37 Get coordinate-wise localization error from scalar noise level 

38 

39 Parameters 

40 ---------- 

41 noise2 : float 

42 the total noise level. In terms of coordinate-wise errors this is 

43 ``noise2 = Δx^2 + Δy^2 + Δz^2`` 

44 pixelsize : array-like 

45 linear extension of pixel/voxel in each direction 

46 

47 Returns 

48 ------- 

49 localization_error : (d,) np.array 

50 localization error distributed to the coordinate directions, 

51 proportionately to the pixel size, i.e. ``Δx``, ``Δy``, ``Δz`` such 

52 that ``noise2 = Δx^2 + Δy^2 + Δz^2``. 

53 """ 

54 voxel = np.asarray(pixelsize) 

55 noise_in_px = np.sqrt(noise2/np.sum(voxel**2)) 

56 return noise_in_px*voxel

58def twoLocusMSD(t, Gamma, tau_c): 

59 """ 

60 MSD for the distance between two loci on an infinite Rouse polymer 

61 

62 Parameters 

63 ---------- 

64 t : iterable 

65 the times at which to evaluate the MSD 

66 Gamma : float 

67 the prefactor for the polymer part (0.5 scaling) of the MSD. Note that 

68 this parameter should be the prefactor for tracking of one locus, i.e. 

69 the MSD produced by this function (which is the distance between two 

70 loci) will have a prefactor of ``2*Gamma``. 

71 tau_e : float 

72 the equilibration time. This is the time for which the two asymptotics 

73 (MSD ~ √t for short times, MSD ~ const at long times) have the same 

74 value. Note that this is a factor π^3/4 ~ 7.75 greater than the Rouse 

75 time of the tether between the loci. 

76 

77 Returns 

78 ------- 

79 (T,) np.array 

80 the MSD evaluated at times `!t` 

81 """ 

82 with np.errstate(divide='ignore'): 

83 ret = 2 * Gamma * ( 

84 np.sqrt(t) * ( 1 - np.exp(-tau_c/(np.pi*t)) ) 

85 + np.sqrt(tau_c) * scipy.special.erfc( np.sqrt(tau_e/(np.pi*t)) ) 

86 ) 

87 ret[np.isnan(ret)] = 0 # t = 0 produces the nan 

88 return ret

90def log_derivative(y, x=None, resampling_density=2): 

91 """ 

92 Calculate loglog-derivative. 

93 

94 We resample the given data to log-spaced x. 

95 

96 Parameters 

97 ---------- 

98 y : array-like 

99 the function values whose derivative we are interested in 

100 x : array-like, optional 

101 the independent variable for the data in y. Will default to 

102 ``np.arange(len(y))`` (and thus ignore the first data point). 

103 resampling_density : float, optional 

104 how tight to space the log-resampled points. A value of 1 corresponds 

105 to the spacing between the first two data points, higher values 

106 decrease spacing. 

107 

108 Returns 

109 ------- 

110 x : np.array 

111 the log-resampled abscissa 

112 dlog : np.array 

113 the calculated log-derivative 

114 """ 

115 if x is None: 

116 x = np.arange(len(y)) 

117 

118 with np.errstate(divide='ignore'): 

119 x = np.log(x) 

120 y = np.log(y) 

121 ind_valid = np.logical_and(np.isfinite(x), np.isfinite(y)) 

122 x = x[ind_valid] 

123 y = y[ind_valid] 

124 

125 dlogx = (x[1] - x[0])/resampling_density 

126 xnew = np.arange(np.min(x), np.max(x), dlogx) 

127 ynew = scipy.interpolate.interp1d(x, y)(xnew) 

128 

129 return np.exp(xnew[:-1] + dlogx/2), np.diff(ynew)/dlogx

131def KM_survival(data, censored, conf=0.95, Tmax=np.inf, S1at=0): 

132 """ 

133 Kaplan-Meier survival estimator on censored data 

134 

135 This is the standard survival estimator for right-censored data, i.e. data 

136 points that are marked as "censored" enter the estimate as ``t_true > t``. 

137 

138 Parameters 

139 ---------- 

140 data : (N,) array-like 

141 individual survival times 

142 censored : (N,) array-like, boolean 

143 indicate for each data point whether it is right-censored or not. 

144 conf : float in (0, 1), optional 

145 the confidence bounds on the survival curve to calculate 

146 Tmax : float 

147 can be used to compute survival only up to some time ``Tmax``. 

148 S1at : float, optional 

149 give a natural lower limit for survival times where S = 1. 

150 

151 Returns 

152 ------- 

153 out : (T, 4) array 

154 the columns of this array are: t, S(t), l(t), u(t), where l and u are 

155 lower and upper confidence levels respectively. 

156 """ 

157 data = np.asarray(data) 

158 censored = np.asarray(censored).astype(bool) 

159 

160 t = np.unique(data[~censored]) # unique also sorts 

161 t = t[t <= Tmax] 

162 S = np.zeros(len(t)+1) 

163 S[0] = 1 

164 V = np.zeros(len(t)+1) 

165 Vsum = 0 

166 for n, curt in enumerate(t, start=1): 

167 d_n = np.count_nonzero(data[~censored] == curt) 

168 N_n = np.count_nonzero(data >= curt) 

169 

170 S[n] = S[n-1]*(1-d_n/N_n) 

171 if N_n > d_n: 

172 Vsum += d_n/(N_n*(N_n-d_n)) 

173 V[n] = np.log(S[n])**(-2)*Vsum 

174 else: 

175 Vsum += np.inf 

176 V[n] = 0 

177 

178 z = scipy.stats.norm().ppf((1-conf)/2) 

179 lower = S**(np.exp( z*np.sqrt(V))) 

180 upper = S**(np.exp(-z*np.sqrt(V))) 

181 

182 if S1at is not None: 

183 t = np.insert(t, 0, S1at) 

184 else: 

185 S = S[1:] 

186 lower = lower[1:] 

187 upper = upper[1:] 

188 

189 return np.stack([t, S, lower, upper], axis=-1)
