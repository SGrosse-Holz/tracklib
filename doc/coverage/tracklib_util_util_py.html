12def _twoLociRelativeACF(ts, A=1, B=1, d=1): # pragma: no cover 

13 """ 

14 A = σ^2 / √κ (general prefactor) 

15 B = Δs^2 / 4κ (tether length) 

16 """

18 def E32(z): 

19 return 2*np.exp(-z) - 2*np.sqrt(np.pi*z)*scipy.special.erfc(np.sqrt(z)) 

20 

21 if not isinstance(ts, collections.abc.Iterable): 

22 ts = [ts]

24 return np.array([ d*A*( np.sqrt(B) - np.sqrt(t/np.pi)*( 1 - 0.5*E32(B/t) ) ) if t != 0 else d*A*np.sqrt(B) for t in ts])

34def distribute_noiselevel(noise2, pixelsize): 

35 """ 

36 Get coordinate-wise localization error from scalar noise level 

37 

38 Parameters 

39 ---------- 

40 noise2 : float 

41 the total noise level. In terms of coordinate-wise errors this is 

42 ``noise2 = Δx^2 + Δy^2 + Δz^2`` 

43 pixelsize : array-like 

44 linear extension of pixel/voxel in each direction 

45 

46 Returns 

47 ------- 

48 localization_error : (d,) np.array 

49 localization error distributed to the coordinate directions, 

50 proportionately to the pixel size, i.e. ``Δx``, ``Δy``, ``Δz`` such 

51 that ``noise2 = Δx^2 + Δy^2 + Δz^2``. 

52 """ 

53 voxel = np.asarray(pixelsize) 

54 noise_in_px = np.sqrt(noise2/np.sum(voxel**2)) 

55 return noise_in_px*voxel

57def twoLocusMSD(t, Gamma, tau_c): 

58 """ 

59 MSD for the distance between two loci on an infinite Rouse polymer 

60 

61 Parameters 

62 ---------- 

63 t : iterable 

64 the times at which to evaluate the MSD 

65 Gamma : float 

66 the prefactor for the polymer part (0.5 scaling) of the MSD. Note that 

67 this parameter should be the prefactor for tracking of one locus, i.e. 

68 the MSD produced by this function (which is the distance between two 

69 loci) will have a prefactor of ``2*Gamma``. 

70 tau_e : float 

71 the equilibration time. This is the time for which the two asymptotics 

72 (MSD ~ √t for short times, MSD ~ const at long times) have the same 

73 value. Note that this is a factor π^3/4 ~ 7.75 greater than the Rouse 

74 time of the tether between the loci. 

75 

76 Returns 

77 ------- 

78 (T,) np.array 

79 the MSD evaluated at times `!t` 

80 """ 

81 with np.errstate(divide='ignore'): 

82 ret = 2 * Gamma * ( 

83 np.sqrt(t) * ( 1 - np.exp(-tau_c/(np.pi*t)) ) 

84 + np.sqrt(tau_c) * scipy.special.erfc( np.sqrt(tau_e/(np.pi*t)) ) 

85 ) 

86 ret[np.isnan(ret)] = 0 # t = 0 produces the nan 

87 return ret

89def log_derivative(y, x=None, resampling_density=2): 

90 """ 

91 Calculate loglog-derivative. 

92 

93 We resample the given data to log-spaced x. 

94 

95 Parameters 

96 ---------- 

97 y : array-like 

98 the function values whose derivative we are interested in 

99 x : array-like, optional 

100 the independent variable for the data in y. Will default to 

101 ``np.arange(len(y))`` (and thus ignore the first data point). 

102 resampling_density : float, optional 

103 how tight to space the log-resampled points. A value of 1 corresponds 

104 to the spacing between the first two data points, higher values 

105 decrease spacing. 

106 

107 Returns 

108 ------- 

109 x : np.array 

110 the log-resampled abscissa 

111 dlog : np.array 

112 the calculated log-derivative 

113 """ 

114 if x is None: 

115 x = np.arange(len(y)) 

116 

117 with np.errstate(divide='ignore'): 

118 x = np.log(x) 

119 y = np.log(y) 

120 ind_valid = np.logical_and(np.isfinite(x), np.isfinite(y)) 

121 x = x[ind_valid] 

122 y = y[ind_valid] 

123 

124 dlogx = (x[1] - x[0])/resampling_density 

125 xnew = np.arange(np.min(x), np.max(x), dlogx) 

126 ynew = scipy.interpolate.interp1d(x, y)(xnew) 

127 

128 return np.exp(xnew[:-1] + dlogx/2), np.diff(ynew)/dlogx
