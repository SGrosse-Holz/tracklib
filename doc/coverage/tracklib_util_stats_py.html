7def KM_survival(data, censored, conf=0.95, Tmax=np.inf, S1at=0): 

8 """ 

9 Kaplan-Meier survival estimator on censored data 

10 

11 This is the standard survival estimator for right-censored data, i.e. data 

12 points that are marked as "censored" enter the estimate as ``t_true > t``. 

13 

14 Parameters 

15 ---------- 

16 data : (N,) array-like 

17 individual survival times 

18 censored : (N,) array-like, boolean 

19 indicate for each data point whether it is right-censored or not. 

20 conf : float in (0, 1), optional 

21 the confidence bounds on the survival curve to calculate 

22 Tmax : float 

23 can be used to compute survival only up to some time ``Tmax``. 

24 S1at : float, optional 

25 give a natural lower limit for survival times where S = 1. 

26 

27 Returns 

28 ------- 

29 (T, 4) array 

30 the columns of this array are: t, S(t), l(t), u(t), where l and u are 

31 lower and upper confidence levels respectively. 

32 """ 

33 data = np.asarray(data) 

34 censored = np.asarray(censored).astype(bool) 

35 

36 t = np.unique(data[~censored]) # unique also sorts 

37 t = t[t <= Tmax] 

38 S = np.zeros(len(t)+1) 

39 S[0] = 1 

40 V = np.zeros(len(t)+1) 

41 Vsum = 0 

42 for n, curt in enumerate(t, start=1): 

43 d_n = np.count_nonzero(data[~censored] == curt) 

44 N_n = np.count_nonzero(data >= curt) 

45 

46 S[n] = S[n-1]*(1-d_n/N_n) 

47 if N_n > d_n: 

48 Vsum += d_n/(N_n*(N_n-d_n)) 

49 V[n] = np.log(S[n])**(-2)*Vsum 

50 else: 

51 Vsum += np.inf 

52 V[n] = 0 

53 

54 z = stats.norm().ppf((1-conf)/2) 

55 lower = S**(np.exp( z*np.sqrt(V))) 

56 upper = S**(np.exp(-z*np.sqrt(V))) 

57 

58 if S1at is not None: 

59 t = np.insert(t, 0, S1at) 

60 else: 

61 S = S[1:] 

62 lower = lower[1:] 

63 upper = upper[1:] 

64 

65 return np.stack([t, S, lower, upper], axis=-1)

67def MLE_censored_exponential(data, censored, conf=0.95): 

68 """ 

69 MLE estimate for exponential distribution, given right-censored data 

70 

71 Parameters 

72 ---------- 

73 data : array-like, dtype=float 

74 the (scalar) values 

75 censored : array-like, dtype=bool, same shape as data 

76 whether the corresponding value in `!data` is censored (``True``) or 

77 completely observed (``False``). 

78 conf : float in [0, 1], optional 

79 the confidence level to use in calculating bounds 

80 

81 Returns 

82 ------- 

83 m, low, high : float 

84 point estimate for the mean of the exponential distribution, and 

85 confidence bounds. 

86 """ 

87 data = np.asarray(data).flatten() 

88 censored = np.asarray(censored, dtype=bool).flatten() 

89 

90 n = np.count_nonzero(~censored) 

91 alpha = 1-conf 

92 

93 # Point estimate 

94 m = np.sum(data) / n 

95 

96 # Confidence interval 

97 c = stats.chi2(1).isf(alpha) / (2*n) 

98 def fitfun(beta): return np.exp(beta) - 1 - beta - c 

99 

100 res = optimize.root_scalar(fitfun, bracket=(-c-1, 0)) 

101 if not res.flag == 'converged': # pragma: no cover 

102 raise RuntimeError("Root finding did not converge for upper confidence interval") 

103 beta_m = res.root 

104 

105 res = optimize.root_scalar(fitfun, bracket=(0, 2*np.sqrt(c))) 

106 if not res.flag == 'converged': # pragma: no cover 

107 raise RuntimeError("Root finding did not converge for lower confidence interval") 

108 beta_p = res.root 

109 

110 return m, m*np.exp(-beta_p), m*np.exp(-beta_m)
